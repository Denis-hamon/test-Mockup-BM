```typescript
import { pgTable, text, timestamp, integer } from "drizzle-orm/pg-core";
import { relations } from "drizzle-orm";
import { users } from "./auth";
import { intakeSubmissions } from "./intake";

export const lawyerProfiles = pgTable("lawyer_profiles", {
  id: text("id").primaryKey().$defaultFn(() => crypto.randomUUID()),
  userId: text("user_id").notNull().unique().references(() => users.id),
  firmName: text("firm_name"),
  phone: text("phone"),
  specialties: text("specialties").notNull().default("[]"), // JSON string[] e.g. ["famille","travail"]
  notifyNewCase: integer("notify_new_case").default(1).notNull(), // 1=on, 0=off
  notifyNewMessage: integer("notify_new_message").default(1).notNull(), // 1=on, 0=off
  createdAt: timestamp("created_at", { mode: "date" }).defaultNow().notNull(),
  updatedAt: timestamp("updated_at", { mode: "date" }).defaultNow().notNull(),
});

export const lawyerNotes = pgTable("lawyer_notes", {
  id: text("id").primaryKey().$defaultFn(() => crypto.randomUUID()),
  submissionId: text("submission_id").notNull().references(() => intakeSubmissions.id),
  lawyerId: text("lawyer_id").notNull().references(() => users.id),
  content: text("content").notNull(),
  createdAt: timestamp("created_at", { mode: "date" }).defaultNow().notNull(),
  updatedAt: timestamp("updated_at", { mode: "date" }).defaultNow().notNull(),
});

// Relations
export const lawyerProfilesRelations = relations(lawyerProfiles, ({ one }) => ({
  user: one(users, { fields: [lawyerProfiles.userId], references: [users.id] }),
}));

export const lawyerNotesRelations = relations(lawyerNotes, ({ one }) => ({
  submission: one(intakeSubmissions, { fields: [lawyerNotes.submissionId], references: [intakeSubmissions.id] }),
  lawyer: one(users, { fields: [lawyerNotes.lawyerId], references: [users.id] }),
}));
```

**1. Create `apps/web/src/server/actions/dashboard.actions.ts`:**

Must be `"use server"` at top. All actions check `session.user.role === "avocat"` (NOT submission ownership like the client-facing getCaseIntelligence). This is the key authorization difference per research pitfall 1.

**listCasesForLawyer(params):**
- Params: `{ page: number; pageSize: number; status?: string; specialty?: string; scoreRange?: "faible" | "moyen" | "eleve"; search?: string; sortBy?: string; sortOrder?: "asc" | "desc" }`
- Auth: `role === "avocat"`, return `{ success: false, error: "unauthorized" }` if not
- Query: SELECT from intakeSubmissions LEFT JOIN qualificationScores (on submissionId) LEFT JOIN caseSummaries (on submissionId)
- WHERE: `status != 'draft'` always. Add filters: if status provided `eq(intakeSubmissions.status, params.status)`, if search provided `ilike(intakeSubmissions.fullName, '%${params.search}%')`, if scoreRange: faible=`overallScore < 40`, moyen=`overallScore >= 40 AND < 70`, eleve=`overallScore >= 70`, if specialty: `eq(caseSummaries.legalDomain, params.specialty)`
- SELECT columns: `id, clientName: fullName, problemType, status, createdAt, overallScore: qualificationScores.overallScore, legalDomain: caseSummaries.legalDomain`
- ORDER BY: default `desc(intakeSubmissions.createdAt)`. If sortBy="score" use `desc(qualificationScores.overallScore)`, if sortBy="status" use `asc(intakeSubmissions.status)`
- LIMIT/OFFSET: `limit(params.pageSize).offset((params.page - 1) * params.pageSize)`
- Count query: same WHERE conditions, `select({ count: sql<number>\`count(*)\` })`
- Return: `{ success: true, data: cases, total: count, page: params.page, pageSize: params.pageSize }`

**getCaseDetailForLawyer(submissionId: string):**
- Auth: `role === "avocat"` check
- Fetch submission (excluding drafts): `db.query.intakeSubmissions.findFirst({ where: and(eq(id, submissionId), sql\`status != 'draft'\`) })`
- If not found: `{ success: false, error: "not_found" }`
- Fetch in parallel: caseSummary, caseTimeline, caseScore, documents (intakeDocuments where submissionId matches), aiFollowUps (where submissionId matches), lawyerNotes (where submissionId matches, ordered by createdAt desc)
- Build CaseIntelligenceResult same way as getCaseIntelligence but without ownership check
- Return: `{ success: true, data: { submission, intelligence: CaseIntelligenceResult, documents: IntakeDocument[], aiFollowUps: AiFollowUp[], notes: LawyerNote[] } }`

**updateCaseStatus(submissionId: string, newStatus: string):**
- Auth: `role === "avocat"`
- Validate newStatus is one of: `"submitted", "en_cours", "termine", "archive"`
- Validate forward-only transitions: define order map `{ submitted: 0, en_cours: 1, termine: 2, archive: 3 }`. Fetch current status, reject if new order <= current order (except submitted->submitted is also rejected). Exception: allow any transition for flexibility in v1 but prevent archive->any-earlier.
- Actually, per research recommendation: allow any forward transition, prevent backward (archive to nouveau). Simplify: just prevent going from `archive` to anything else. All other transitions allowed.
- `db.update(intakeSubmissions).set({ status: newStatus, updatedAt: new Date() }).where(eq(intakeSubmissions.id, submissionId))`
- Return `{ success: true }`

**addLawyerNote(submissionId: string, content: string):**
- Auth: `role === "avocat"`, get `session.user.id` as lawyerId
- Insert into lawyerNotes: `{ submissionId, lawyerId: session.user.id, content }`
- Return `{ success: true, noteId }`

**updateLawyerNote(noteId: string, content: string):**
- Auth: `role === "avocat"`
- Verify note ownership: `lawyerNotes.lawyerId === session.user.id`
- Update content and updatedAt
- Return `{ success: true }`

**deleteLawyerNote(noteId: string):**
- Auth: `role === "avocat"`, verify ownership
- Delete note
- Return `{ success: true }`

**regenerateCaseForLawyer(submissionId: string):**
- Auth: `role === "avocat"`
- Verify submission exists and is not draft
- Call `triggerCaseIntelligence(submissionId)` from case-intelligence.actions.ts
- Return result
