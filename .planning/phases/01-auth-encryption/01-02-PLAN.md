    **Auth.js config** (`apps/web/src/lib/auth.ts`):
    Per D-05 (30-day sessions), D-07 (roles in JWT), research Pattern 1:
    ```typescript
    import NextAuth from "next-auth";
    import { DrizzleAdapter } from "@auth/drizzle-adapter";
    import Credentials from "next-auth/providers/credentials";
    import { db } from "@/lib/db";
    import { users } from "@/lib/db/schema/auth";
    import { loginSchema } from "@legalconnect/shared";
    import bcrypt from "bcryptjs";
    import { eq } from "drizzle-orm";

    export const { handlers, auth, signIn, signOut } = NextAuth({
      adapter: DrizzleAdapter(db),
      session: {
        strategy: "jwt",
        maxAge: 30 * 24 * 60 * 60, // 30 days per D-05
      },
      providers: [
        Credentials({
          async authorize(credentials) {
            const { email, password } = loginSchema.parse(credentials);
            const user = await db.query.users.findFirst({
              where: eq(users.email, email),
            });
            if (!user || !user.passwordHash) return null;
            if (!user.emailVerified) return null;
            if (user.deletedAt) return null; // Soft-deleted users cannot log in
            const valid = await bcrypt.compare(password, user.passwordHash);
            if (!valid) return null;
            return { id: user.id, email: user.email, role: user.role, name: user.name };
          },
        }),
      ],
      callbacks: {
        jwt({ token, user }) {
          if (user) {
            token.role = user.role;
          }
          return token;
        },
        session({ session, token }) {
          session.user.role = token.role as "avocat" | "client";
          session.user.id = token.sub!;
          return session;
        },
      },
      pages: {
        signIn: "/login",
        newUser: "/register",
      },
    });
    ```

    **Proxy** (`apps/web/proxy.ts`):
    Per research Pattern 2 (Next.js 16 proxy.ts replaces middleware.ts):
    ```typescript
    import { auth } from "@/lib/auth";
    import { NextResponse } from "next/server";

    export default auth((req) => {
      const isLoggedIn = !!req.auth;
      const isAuthPage = req.nextUrl.pathname.startsWith("/login") ||
        req.nextUrl.pathname.startsWith("/register") ||
        req.nextUrl.pathname.startsWith("/verify-email") ||
        req.nextUrl.pathname.startsWith("/reset-password");
      const isApiAuth = req.nextUrl.pathname.startsWith("/api/auth");

      if (isApiAuth) return NextResponse.next();
      if (isAuthPage && isLoggedIn) {
        return NextResponse.redirect(new URL("/dashboard", req.nextUrl));
      }
      if (!isLoggedIn && !isAuthPage) {
        return NextResponse.redirect(new URL("/login", req.nextUrl));
      }
    });

    export const config = {
      matcher: ["/((?!api/auth|_next/static|_next/image|favicon.ico).*)"],
    };
    ```

    **Registration server action** (`apps/web/src/server/actions/auth.actions.ts`):
    Per D-07 (role selection), D-08 (8 char min via Zod), D-14 (24h verification, 1h reset):
    - `registerUser(formData)`: Parse with registerSchema, check existing user, hash password with bcrypt (cost 12), insert user, create emailVerificationToken (24h expiry per D-14), send verification email via Resend (D-12 warm tone, D-13 sender). Return `{ success: true }` or `{ error: string }`.
    - `verifyEmail(token)`: Find token, check not expired (24h per D-14), check not used, mark emailVerified on user, mark token as used. Return `{ success: true }` or `{ error: string }`.
    - `requestPasswordReset(formData)`: Parse with resetPasswordRequestSchema, find user, create passwordResetToken (1h expiry per D-14), send reset email. Always return `{ success: true }` (don't reveal if email exists).
    - `resetPassword(formData)`: Parse with resetPasswordSchema, find token, check not expired/used, hash new password with bcrypt (cost 12), update user.passwordHash, mark token used. Return `{ success: true }` or `{ error: string }`.

    All server actions must use `"use server"` directive at the top.
